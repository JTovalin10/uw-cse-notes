$$ \bullet ; \text{size}(\bullet) := 1 $$

$$ \bullet ; \text{size}\left(\begin{array}{c} \bullet \ / ; \backslash \ T_1 ; T_2 \end{array}\right) := 1 + \text{size}(T_1) + \text{size}(T_2) $$

$$ \bullet ; \text{height}(\bullet) := 0 $$

$$ \bullet \text{height}\left(\begin{array}{c} \bullet \ / ; \backslash \ T_1 ; T_2 \end{array}\right) := 1 + \max{\text{height}(T_1), \text{height}(T_2)} $$
